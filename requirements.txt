name: Monitor T-cup 2026

on:
  # Spouštění podle časového plánu - výchozí je každých 15 minut.
  # Chcete-li změnit interval, upravte řádek "cron" níže.
  # Formát: minuta hodina den měsíc den-v-týdnu (čas je v UTC!)
  schedule:
    - cron: "*/15 * * * *"

  # Umožňuje spustit kontrolu ručně tlačítkem v záložce "Actions" na GitHubu
  # (užitečné pro otestování, že vše funguje).
  workflow_dispatch: {}

# Aby skript mohl ukládat stav zpět do repozitáře (commit + push),
# potřebuje oprávnění zapisovat do obsahu repozitáře.
permissions:
  contents: write

# Zajistí, že se nebudou překrývat dva běhy najednou (např. pokud by
# jeden běh trval déle než 15 minut).
concurrency:
  group: tcup-monitor
  cancel-in-progress: false

jobs:
  check:
    runs-on: ubuntu-latest
    steps:
      - name: Stažení repozitáře
        uses: actions/checkout@v4
        with:
          persist-credentials: true

      - name: Instalace Pythonu
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Instalace závislostí
        run: pip install -r requirements.txt

      - name: Spuštění kontroly
        env:
          SMTP_USER: ${{ secrets.SMTP_USER }}
          SMTP_PASSWORD: ${{ secrets.SMTP_PASSWORD }}
          EMAIL_TO: ${{ secrets.EMAIL_TO }}
        run: python monitor.py

      - name: Uložení nového stavu do repozitáře
        run: |
          git config user.name "tcup-monitor-bot"
          git config user.email "actions@users.noreply.github.com"
          git add state/
          git diff --quiet --cached || git commit -m "Aktualizace stavu monitoru [skip ci]"
          git push
